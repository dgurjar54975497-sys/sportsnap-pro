async function loadMatches() { const box = document.getElementById("matches"); box.innerHTML = "Loading..."; try { const res = await fetch("/api/live"); const data = await res.json(); if (!data || data.length === 0) { box.innerHTML = "⚠️ No matches found"; return; } let html = ""; data.forEach(m => { html += `

${m.title}

${m.league}

${m.date}

`; }); box.innerHTML = html; } catch (err) { box.innerHTML = "❌ Error loading data"; } } loadMatches();
